#### **Enhanced CSV Generation**
```typescript
private generateOptimizedCSV(matchData: MatchData): File {
  // Filter relevant events only
  const relevantEvents = matchData.events.filter(event => 
    ['goal', 'assist', 'turnover', 'foul'].includes(event.type)
  );
  
  const headers = ['Time', 'Event', 'Team', 'Details'];
  const rows = relevantEvents.map((event: GameEvent) => [
    event.gameTime || '',
    event.type || '',
    event.team === 'us' ? 'Us' : 'Them',
    event.label || ''
  ]);

  const csvContent = [headers, ...rows]
    .map(row => row.map(cell => `"${cell}"`).join(','))
    .join('\n');

  return new File([csvContent], 'track-side-match-data.csv', { type: 'text/csv' });
}
```

          {displayTime}
        </div>
      </div>

      {/* Clear, functional timer control */}
      <button
        onClick={toggleTimer}
        className={`timer-control px-4 py-2 rounded-lg font-semibold transition-all ${

            ? 'bg-green-600 hover:bg-green-700 text-white' 
            : 'bg-red-600 hover:bg-red-700 text-white animate-pulse'
        }`}
      >
        {isRunning ? 'Pause' : 'Start'}
      </button>
    </div>
